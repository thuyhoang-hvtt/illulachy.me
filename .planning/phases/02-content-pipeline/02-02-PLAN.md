  <action>
    1. Install chokidar for file watching:

    2. Create `src/vite-plugin-timeline.ts`:
       ```typescript
       import { Plugin } from 'vite'
       import { spawn } from 'child_process'
       import chokidar from 'chokidar'
       
       export function timelinePlugin(): Plugin {
         return {
           name: 'timeline-generator',
           
           // Run generator on build start
           async buildStart() {
             console.log('[Timeline Plugin] Generating timeline.json...')
             await runGenerator()
           },
           
           // Watch content files during dev
           configureServer(server) {
             const watcher = chokidar.watch('content/**/*.md', {
               ignoreInitial: true,
               awaitWriteFinish: { stabilityThreshold: 100 }
             })
             
             const handleChange = async (path: string) => {
               console.log(`[Timeline Plugin] Content changed: ${path}`)
               try {
                 await runGenerator()
                 server.ws.send({ type: 'full-reload' })
               } catch (error) {
                 server.ws.send({
                   type: 'error',
                   err: { message: String(error) }
                 })
               }
             }
             
             watcher.on('add', handleChange)
             watcher.on('change', handleChange)
             watcher.on('unlink', handleChange)
             
             return () => watcher.close()
           }
         }
       }
       
       function runGenerator(): Promise<void> {
         return new Promise((resolve, reject) => {
           const proc = spawn('npm', ['run', 'generate-timeline'], {
             stdio: 'inherit',
             shell: true
           })
           proc.on('exit', (code) => {
             code === 0 ? resolve() : reject(new Error(`Generator failed with code ${code}`))
           })
           proc.on('error', reject)
         })
       }
       ```
    
    3. Update `vite.config.ts` to use the plugin:
       ```typescript
       import { timelinePlugin } from './src/vite-plugin-timeline'
       
       export default defineConfig({
         plugins: [react(), timelinePlugin()],
         // ... existing config
       })
       ```
    
    4. Test that `npm run build` now runs the generator first
  </action>

    <automated>npm run build 2>&1 | grep -E "(Timeline|timeline)" && test -f public/timeline.json</automated>
  </verify>
  <done>Vite plugin registered; `npm run build` generates timeline.json before bundling</done>
